    30→} from 'lucide-react'

    32→// ─── প্রকারভেদ ───
    33→interface ConversationWithContact extends ConversationItem {
    34→  contact: NonNullable<ConversationItem['contact']>
    35→}
    36→
    37→type FilterTab = 'all' | 'unread' | 'open' | 'assigned' | 'mine' | 'has_ticket'
    38→
    39→// ─── ট্যাব গণনা সহায়ক ───
    40→interface TabCounts {
    41→  all: number
    42→  unread: number
    43→  open: number
    44→  assigned: number
    45→  mine: number
    46→  has_ticket: number
    47→}
    48→
    49→// ─── প্রধান উপাদান ───
    50→export default function WhatsAppComplaint() {
    51→  const { user, pageParams, navigate } = useAppStore()
    52→  const { toast } = useToast()
    53→  const queryClient = useQueryClient()
    54→
    55→  // স্থানীয় অবস্থা
    56→  const [searchQuery, setSearchQuery] = useState('')
    57→  const [activeFilter, setActiveFilter] = useState<FilterTab>('all')
    58→  const [selectedConversationId, setSelectedConversationId] = useState<string | null>(
    59→    pageParams?.conversationId || null
    60→  )
    61→  const [replyingTo, setReplyingTo] = useState<WhatsAppMessage | null>(null)
    62→  const [showRightPanel, setShowRightPanel] = useState(true)
    63→  const [mobileView, setMobileView] = useState<'list' | 'chat'>('list')
    64→  const [detailSheetOpen, setDetailSheetOpen] = useState(false)
    65→
    66→  const messagesEndRef = useRef<HTMLDivElement>(null)
    67→  const messagesContainerRef = useRef<HTMLDivElement>(null)
    68→
    69→  // ─── সকেট ───
    70→  const {
    71→    isConnected,
    72→    joinConversation,
    73→    leaveConversation,
    74→    emitTyping,
    75→  } = useWhatsAppSocket({
    76→    onNewMessage: (data: any) => {
    77→      queryClient.invalidateQueries({ queryKey: ['wa-conversations'] })
    78→      if (data.conversationId === selectedConversationId) {
    79→        queryClient.invalidateQueries({ queryKey: ['wa-messages', selectedConversationId] })
    80→      }
    81→    },
    82→    onConversationUpdated: () => {
    83→      queryClient.invalidateQueries({ queryKey: ['wa-conversations'] })
    84→      if (selectedConversationId) {
    85→        queryClient.invalidateQueries({ queryKey: ['wa-conversation', selectedConversationId] })
    86→      }
    87→    },
    88→    onNotification: (data: any) => {
    89→      toast({ title: data.title || 'New notification', description: data.body })
    90→    },
    91→  })
    92→
    93→  // ─── কথোপকথন রুমে যোগদান/বাহির ───
    94→  useEffect(() => {
    95→    if (selectedConversationId && isConnected) {
    96→      joinConversation(selectedConversationId)
    97→    }
    98→    return () => {
    99→      if (selectedConversationId) {
   100→        leaveConversation(selectedConversationId)
   101→      }
   102→    }
   103→  }, [selectedConversationId, isConnected])
   104→
   105→  // ─── কথোপকথন আনা ───
   106→  const { data: conversations = [], isLoading: convLoading } = useQuery<ConversationWithContact[]>({
   107→    queryKey: ['wa-conversations', activeFilter, searchQuery],
   108→    queryFn: async () => {
   109→      const params = new URLSearchParams()
   110→      if (searchQuery) params.set('search', searchQuery)
   111→      params.set('filter', activeFilter)
   112→      const res = await api.get<ConversationWithContact[]>(`/api/maintenance/whatsapp/conversations?${params}`)
   113→      return res.success ? (res.data || []) : []
   114→    },
   115→    refetchInterval: 15000,
   116→  })
   117→
   118→  // ─── নির্বাচিত কথোপকথনের বিবরণ আনা ───
   119→  const { data: conversationDetail, isLoading: detailLoading } = useQuery({
   120→    queryKey: ['wa-conversation', selectedConversationId],
   121→    queryFn: async () => {
   122→      if (!selectedConversationId) return null
   123→      const res = await api.get(`/api/maintenance/whatsapp/conversations/${selectedConversationId}`)
   124→      return res.success ? res.data : null
   125→    },
   126→    enabled: !!selectedConversationId,
   127→  })
   128→
   129→  // ─── বার্তা আনা ───
   130→  const { data: messages = [], isLoading: msgsLoading } = useQuery<WhatsAppMessage[]>({
   131→    queryKey: ['wa-messages', selectedConversationId],
   132→    queryFn: async () => {
   133→      if (!selectedConversationId) return []
   134→      const res = await api.get<WhatsAppMessage[]>(`/api/maintenance/whatsapp/conversations/${selectedConversationId}/messages`)
   135→      return res.success ? (res.data || []) : []
   136→    },
   137→    enabled: !!selectedConversationId,
   138→  })
   139→
   140→  // ─── বার্তা পাঠানো মিউটেশন ───
   141→  const sendMessageMutation = useMutation({
   142→    mutationFn: async ({ content, type, file }: { content: string; type: string; file?: File }) => {
   143→      if (file) {
   144→        const formData = new FormData()
   145→        formData.append('file', file)
   146→        formData.append('content', content)
   147→        formData.append('type', type)
   148→        if (replyingTo) formData.append('repliedToId', replyingTo.id)
   149→        const res = await fetch(`/api/maintenance/whatsapp/conversations/${selectedConversationId}/messages?XTransformPort=3006`, {
   150→          method: 'POST',
   151→          headers: { Authorization: `Bearer ${useAppStore.getState().token}` },
   152→          body: formData,
   153→        })
   154→        return res.json()
   155→      }
   156→      return api.post(`/api/maintenance/whatsapp/conversations/${selectedConversationId}/messages`, {
   157→        content,
   158→        type,
   159→        repliedToId: replyingTo?.id,
   160→      })
   161→    },
   162→    onSuccess: () => {
   163→      queryClient.invalidateQueries({ queryKey: ['wa-messages', selectedConversationId] })
   164→      queryClient.invalidateQueries({ queryKey: ['wa-conversations'] })
   165→      setReplyingTo(null)
   166→    },
   167→    onError: (err: any) => {
   168→      toast({ title: err.message || 'Failed to send message', variant: 'destructive' })
   169→    },
   170→  })
   171→
   172→  // ─── পঠিত হিসেবে চিহ্নিত ───
   173→  const markReadMutation = useMutation({
   174→    mutationFn: async (convId: string) => {
   175→      return api.put(`/api/maintenance/whatsapp/conversations/${convId}/read`, {})
   176→    },
   177→    onSuccess: () => {
   178→      queryClient.invalidateQueries({ queryKey: ['wa-conversations'] })
   179→      queryClient.invalidateQueries({ queryKey: ['wa-conversation', selectedConversationId] })
   180→    },
   181→  })
   182→
   183→  // ─── নির্বাচনে স্বয়ংক্রিয়ভাবে পঠিত হিসেবে চিহ্নিত ───
   184→  useEffect(() => {
   185→    if (selectedConversationId) {
   186→      const conv = conversations.find(c => c.id === selectedConversationId)
   187→      if (conv && conv.unreadCount > 0) {
   188→        markReadMutation.mutate(selectedConversationId)
   189→      }
   190→    }
   191→  }, [selectedConversationId])
   192→
   193→  // ─── নতুন বার্তায় নিচে স্ক্রল ───
   194→  useEffect(() => {
   195→    if (messages.length > 0) {
   196→      messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' })
   197→    }
   198→  }, [messages.length])
   199→
   200→  // ─── গণনাকৃত ট্যাব গণনা ───
   201→  const tabCounts = useMemo<TabCounts>(() => {
   202→    const counts: TabCounts = { all: conversations.length, unread: 0, open: 0, assigned: 0, mine: 0, has_ticket: 0 }
   203→    for (const c of conversations) {
   204→      if (c.unreadCount > 0) counts.unread++
   205→      if (c.status === 'open') counts.open++
   206→      if (c.assignedToId) counts.assigned++
   207→      if (c.assignedToId === user?.id) counts.mine++
   208→      if (c.ticketId) counts.has_ticket++
   209→    }
   210→    return counts
   211→  }, [conversations, user?.id])
   212→
   213→  // ─── নির্বাচিত কথোপকথনের তথ্য ───
   214→  const selectedConversation = conversations.find(c => c.id === selectedConversationId)
   215→  const contactData = conversationDetail?.contact || selectedConversation?.contact
   216→
   217→  // ─── তারিখ অনুযায়ী বার্তা গোষ্ঠীবদ্ধ করা ───
   218→  const groupedMessages = useMemo(() => {
   219→    const groups: { date: string; messages: WhatsAppMessage[] }[] = []
   220→    let currentDate = ''
   221→
   222→    for (const msg of messages) {
   223→      const msgDate = new Date(msg.createdAt).toDateString()
   224→      if (msgDate !== currentDate) {
   225→        currentDate = msgDate
   226→        groups.push({ date: msg.createdAt, messages: [msg] })
   227→      } else {
   228→        groups[groups.length - 1].messages.push(msg)
   229→      }
   230→    }
   231→    return groups
   232→  }, [messages])
   233→
   234→  // ─── হ্যান্ডলারসমূহ ───
   235→  const handleSelectConversation = (id: string) => {
   236→    setSelectedConversationId(id)
   237→    setReplyingTo(null)
   238→    setMobileView('chat')
   239→  }
   240→
   241→  const handleSendMessage = (content: string, type: 'text' | 'image' | 'document' | 'video' | 'audio', file?: File) => {
   242→    sendMessageMutation.mutate({ content, type, file })
   243→  }
   244→
   245→  const handleBackToList = () => {
   246→    setMobileView('list')
   247→  }
   248→
   249→  const handleRefresh = () => {
   250→    queryClient.invalidateQueries({ queryKey: ['wa-conversations'] })
   251→    queryClient.invalidateQueries({ queryKey: ['wa-conversation', selectedConversationId] })
   252→    queryClient.invalidateQueries({ queryKey: ['wa-messages', selectedConversationId] })
   253→  }
   254→
   255→  const handleNavigateToTicket = (ticketId: string) => {
   256→    navigate('maintenance-ticket-detail', { id: ticketId })
   257→  }
   258→
   259→  const handleNavigateToCustomer = (customerId: string) => {
   260→    navigate('customers', { id: customerId })
   261→  }
   262→
   263→  const handleStarMessage = (messageId: string) => {
   264→    api.put(`/api/maintenance/whatsapp/messages/${messageId}`, { isStarred: true }).then(() => {
   265→      queryClient.invalidateQueries({ queryKey: ['wa-messages', selectedConversationId] })
   266→    })
   267→  }
   268→
   269→  // ─── ফাঁকা চ্যাট অবস্থা ───
   270→  const renderEmptyChat = () => (
   271→    <div className="flex-1 flex items-center justify-center bg-gray-50 dark:bg-gray-900/50">
   272→      <div className="text-center space-y-3 px-6">
   273→        <div className="w-20 h-20 mx-auto rounded-full bg-emerald-100 dark:bg-emerald-900/30 flex items-center justify-center">
   274→          <MessageSquare className="h-9 w-9 text-emerald-600 dark:text-emerald-400" />
   275→        </div>
   276→        <div>
   277→          <h3 className="text-lg font-semibold text-gray-700 dark:text-gray-300">WhatsApp Inbox</h3>
   278→          <p className="text-sm text-gray-500 mt-1">
   279→            Select a conversation from the left panel to start chatting with customers.
   280→          </p>
   281→        </div>
   282→        {!isConnected && (
   283→          <div className="flex items-center justify-center gap-1.5 text-xs text-red-500">
   284→            <WifiOff className="h-3.5 w-3.5" />
   285→            Real-time disconnected
   286→          </div>
   287→        )}
   288→      </div>
   289→    </div>
   290→  )
   291→
   292→  // ─── চ্যাট লোডিং স্কেলেটন ───
   293→  const renderChatSkeleton = () => (
   294→    <div className="flex-1 flex flex-col bg-white dark:bg-gray-950">
   295→      <Skeleton className="h-16 border-b border-gray-200 dark:border-gray-800" />
   296→      <div className="flex-1 p-4 space-y-4">
   297→        {[...Array(6)].map((_, i) => (
   298→          <div key={i} className={cn('flex gap-2', i % 2 === 0 ? 'justify-start' : 'justify-end')}>
   299→            <Skeleton className={cn('h-16 rounded-2xl', i % 2 === 0 ? 'w-3/5' : 'w-2/5')} />
   300→          </div>
   301→        ))}
   302→      </div>
   303→    </div>
   304→  )
   305→
   306→  // ─── চ্যাট দৃশ্য ───
   307→  const renderChatView = () => {
   308→    if (!selectedConversationId || !selectedConversation) return renderEmptyChat()
   309→    if (msgsLoading && messages.length === 0) return renderChatSkeleton()
   310→
   311→    const contactName = selectedConversation.contact?.name || selectedConversation.contact?.pushName || selectedConversation.contact?.phone || 'Unknown'
   312→
   313→    return (
   314→      <div className="flex-1 flex flex-col bg-white dark:bg-gray-950 min-w-0">
   315→        {/* চ্যাট হেডার */}
   316→        <div className="h-16 flex items-center justify-between px-4 border-b border-gray-200 dark:border-gray-800 bg-white dark:bg-gray-950 flex-shrink-0">
   317→          <div className="flex items-center gap-3 min-w-0">
   318→            {/* মোবাইল ব্যাক বাটন */}
   319→            <Button
   320→              variant="ghost"
   321→              size="icon"
   322→              className="lg:hidden h-9 w-9 flex-shrink-0"
   323→              onClick={handleBackToList}
   324→            >
   325→              <ArrowLeft className="h-5 w-5" />
   326→            </Button>
   327→
   328→            <Avatar className="h-10 w-10 flex-shrink-0">
   329→              {selectedConversation.contact?.profilePicUrl && (
   330→                <img src={selectedConversation.contact.profilePicUrl} alt={contactName} className="object-cover w-full h-full rounded-full" />
   331→              )}
   332→              <AvatarFallback className="text-xs bg-gray-100 dark:bg-gray-800 text-gray-600 dark:text-gray-300">
   333→                {(selectedConversation.contact?.name || selectedConversation.contact?.phone || '??').substring(0, 2).toUpperCase()}
   334→              </AvatarFallback>
   335→            </Avatar>
   336→            <div className="min-w-0">
   337→              <h3 className="text-sm font-semibold text-gray-900 dark:text-gray-100 truncate">{contactName}</h3>
   338→              <p className="text-xs text-gray-500 truncate">{selectedConversation.contact?.phone}</p>
   339→            </div>
   340→          </div>
   341→
   342→          <div className="flex items-center gap-1 flex-shrink-0">
   343→            {/* সংযোগ সূচক */}
   344→            <Tooltip>
   345→              <TooltipTrigger>
   346→                {isConnected ? (
   347→                  <Wifi className="h-4 w-4 text-emerald-500" />
   348→                ) : (
   349→                  <WifiOff className="h-4 w-4 text-red-400" />
   350→                )}
   351→              </TooltipTrigger>
   352→              <TooltipContent>
   353→                {isConnected ? 'Connected' : 'Disconnected'}
   354→              </TooltipContent>
   355→            </Tooltip>
   356→
   357→            {/* পঠিত হিসেবে চিহ্নিত */}
   358→            {selectedConversation.unreadCount > 0 && (
   359→              <Tooltip>
   360→                <TooltipTrigger asChild>
   361→                  <Button variant="ghost" size="icon" className="h-8 w-8" onClick={() => markReadMutation.mutate(selectedConversationId)}>
   362→                    <CheckCheck className="h-4 w-4 text-emerald-600" />
   363→                  </Button>
   364→                </TooltipTrigger>
   365→                <TooltipContent>Mark as read</TooltipContent>
   366→              </Tooltip>
   367→            )}
   368→
   369→            {/* ডান প্যানেল টগল */}
   370→            <Tooltip>
   371→              <TooltipTrigger asChild>
   372→                <Button
   373→                  variant="ghost"
   374→                  size="icon"
   375→                  className="h-8 w-8 hidden lg:flex"
   376→                  onClick={() => setShowRightPanel(!showRightPanel)}
   377→                >
   378→                  {showRightPanel ? <PanelRightClose className="h-4 w-4" /> : <PanelRightOpen className="h-4 w-4" />}
   379→                </Button>
   380→              </TooltipTrigger>
   381→              <TooltipContent>{showRightPanel ? 'Hide details' : 'Show details'}</TooltipContent>
   382→            </Tooltip>
   383→
   384→            {/* তথ্য / মোবাইলে বিস্তারিত শিট খোলা */}
   385→            <Tooltip>
   386→              <TooltipTrigger asChild>
   387→                <Button
   388→                  variant="ghost"
   389→                  size="icon"
   390→                  className="h-8 w-8 lg:hidden"
   391→                  onClick={() => setDetailSheetOpen(true)}
   392→                >
   393→                  <Info className="h-4 w-4" />
   394→                </Button>
   395→              </TooltipTrigger>
   396→              <TooltipContent>Details</TooltipContent>
   397→            </Tooltip>
   398→
   399→            {/* আরো কার্যকলাপ */}
   400→            <DropdownMenu>
   401→              <DropdownMenuTrigger asChild>
   402→                <Button variant="ghost" size="icon" className="h-8 w-8">
   403→                  <MoreVertical className="h-4 w-4" />
   404→                </Button>
   405→              </DropdownMenuTrigger>
   406→              <DropdownMenuContent align="end" className="w-48">
   407→                <DropdownMenuItem onClick={() => markReadMutation.mutate(selectedConversationId)}>
   408→                  <CheckCheck className="h-4 w-4 mr-2" />
   409→                  Mark as Read
   410→                </DropdownMenuItem>
   411→                <DropdownMenuItem>
   412→                  <Star className="h-4 w-4 mr-2" />
   413→                  Star Conversation
   414→                </DropdownMenuItem>
   415→                <DropdownMenuSeparator />
   416→                <DropdownMenuItem className="text-red-600">
   417→                  <Archive className="h-4 w-4 mr-2" />
   418→                  Archive Chat
   419→                </DropdownMenuItem>
   420→              </DropdownMenuContent>
   421→            </DropdownMenu>
   422→          </div>
   423→        </div>
   424→
   425→        {/* বার্তা এলাকা */}
   426→        <div
   427→          ref={messagesContainerRef}
   428→          className="flex-1 overflow-y-auto px-3 py-3 bg-[#efeae2] dark:bg-gray-900 space-y-0.5"
   429→        >
   430→          {messages.length === 0 && !msgsLoading && (
   431→            <div className="flex items-center justify-center h-full">
   432→              <p className="text-sm text-gray-400">No messages yet</p>
   433→            </div>
   434→          )}
   435→
   436→          {groupedMessages.map((group) => (
   437→            <React.Fragment key={group.date}>
   438→              <DateSeparator date={group.date} />
   439→              {group.messages.map((msg) => (
   440→                <WhatsAppMessageBubble
   441→                  key={msg.id}
   442→                  message={msg}
   443→                  onReply={(m) => setReplyingTo(m)}
   444→                  onStar={handleStarMessage}
   445→                />
   446→              ))}
   447→            </React.Fragment>
   448→          ))}
   449→
   450→          <div ref={messagesEndRef} />
   451→        </div>
   452→
   453→        {/* বার্তা ইনপুট */}
   454→        <WhatsAppMessageInput
   455→          onSend={handleSendMessage}
   456→          onTyping={() => emitTyping(selectedConversationId)}
   457→          replyingTo={replyingTo}
   458→          onCancelReply={() => setReplyingTo(null)}
   459→          disabled={sendMessageMutation.isPending}
   460→        />
   461→      </div>
   462→    )
   463→  }
   464→
   465→  // ─── প্রধান বিন্যাস ───
   466→  return (
   467→    <div className="h-[calc(100vh-3.5rem)] flex flex-col bg-background">
   468→      {/* সংযোগ স্ট্যাটাস ও রিফ্রেশসহ শীর্ষ বার */}
   469→      <div className="h-11 flex items-center justify-between px-4 border-b border-gray-200 dark:border-gray-800 bg-white dark:bg-gray-950 flex-shrink-0">
   470→        <div className="flex items-center gap-2">
   471→          <MessageSquare className="h-4 w-4 text-emerald-600" />
