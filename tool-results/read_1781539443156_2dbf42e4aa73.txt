     3→function daysAgo(days: number): Date {
     4→  const d = new Date()
     5→  d.setDate(d.getDate() - days)
     6→  d.setHours(0, 0, 0, 0)
     7→  return d
     8→}
     9→
    10→function daysFromNow(days: number): Date {
    11→  const d = new Date()
    12→  d.setDate(d.getDate() + days)
    13→  d.setHours(0, 0, 0, 0)
    14→  return d
    15→}
    16→
    17→function daysFromNowTime(days: number): Date {
    18→  const d = new Date()

    20→  return d
