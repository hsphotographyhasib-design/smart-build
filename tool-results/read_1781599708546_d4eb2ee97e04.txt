   696‚Üí   752‚Üí/**
   697‚Üí   753‚Üí * Check if a user with the given role can access a specific API route with the given method.
   698‚Üí   754‚Üí * Returns true if access is allowed.
   699‚Üí   755‚Üí */
   700‚Üí   756‚Üíexport function canAccessRoute(role: string, pathname: string, method: string): boolean {
   701‚Üí   757‚Üí  // Super admin has access to everything
   702‚Üí   758‚Üí  if (role === ROLES.SUPER_ADMIN) return true
   703‚Üí   759‚Üí
   704‚Üí   760‚Üí  // Find the most specific matching permission
   705‚Üí   761‚Üí  let matched = false
   706‚Üí   762‚Üí  for (const perm of ROUTE_PERMISSIONS) {

   711‚Üí   767‚Üí    if (!allowedRoles) continue
   712‚Üí   768‚Üí
   713‚Üí   769‚Üí    // '*' means any authenticated user

   718‚Üí   774‚Üí
   719‚Üí   775‚Üí    // Check if the user's role is in the allowed list
   720‚Üí   776‚Üí    if (allowedRoles.includes(role)) {
   721‚Üí   777‚Üí      matched = true
   722‚Üí   778‚Üí      break
   723‚Üí   779‚Üí    }
   724‚Üí   780‚Üí  }

   729‚Üí   785‚Üí/**
   730‚Üí   786‚Üí * Match a URL pathname against a route pattern like "/api/projects/[id]"
   731‚Üí   787‚Üí */

   734‚Üí   790‚Üí  if (pathname === pattern) return true
   735‚Üí   791‚Üí

   741‚Üí   797‚Üí

   745‚Üí   801‚Üí

   748‚Üí   804‚Üí      continue

   755‚Üí   811‚Üí  }
   756‚Üí   812‚Üí
