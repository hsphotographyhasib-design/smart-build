   103→// ─── SLA কাউন্টডাউন কম্পোনেন্ট ───
   104→function SLACountdown({ deadline }: { deadline: string }) {
   105→  const [remaining, setRemaining] = useState('')
   106→
   107→  useEffect(() => {
   108→    function calc() {
   109→      const diff = new Date(deadline).getTime() - Date.now()
   110→      if (diff <= 0) {
   111→        setRemaining('BREACHED')
   112→        return
   113→      }
   114→      const hours = Math.floor(diff / 3600000)
   115→      const mins = Math.floor((diff % 3600000) / 60000)
   116→      if (hours > 24) {
   117→        const days = Math.floor(hours / 24)
   118→        setRemaining(`${days}d ${hours % 24}h`)
   119→      } else {
   120→        setRemaining(`${hours}h ${mins}m`)
   121→      }
   122→    }
   123→    calc()
   124→    const interval = setInterval(calc, 60000)
   125→    return () => clearInterval(interval)
   126→  }, [deadline])
   127→
   128→  const isBreached = remaining === 'BREACHED'
   129→
   130→  return (
   131→    <span className={cn(
   132→      'text-xs font-mono font-medium',
   133→      isBreached ? 'text-red-600 dark:text-red-400' : 'text-amber-600 dark:text-amber-400'
   134→    )}>
   135→      {isBreached && '⚠ '}{remaining}
   136→    </span>
   137→  )
   138→}

   141→function WorkflowStepper({ status }: { status: string }) {
   142→  const currentStep = STATUS_CONFIG[status]?.step ?? -1
   143→
   144→  return (
   145→    <div>
   146→      <div className="flex items-center gap-0.5">
   147→        {STATUS_WORKFLOW.map((step, idx) => {
   148→          const isActive = currentStep >= idx
   149→          const isCurrent = currentStep === idx
   150→          return (
   151→            <React.Fragment key={step}>
   152→              <div
   153→                className={cn(
   154→                  'h-5 w-5 rounded-full flex items-center justify-center text-[9px] font-bold transition-colors shrink-0',
   155→                  isActive
   156→                    ? isCurrent
   157→                      ? 'bg-emerald-500 text-white ring-2 ring-emerald-200 dark:ring-emerald-800'
   158→                      : 'bg-emerald-500 text-white'
   159→                    : 'bg-muted text-muted-foreground'
   160→                )}
   161→                title={STATUS_CONFIG[step]?.label}
   162→              >
   163→                {isActive ? <CheckCircle2 className="h-3 w-3" /> : idx + 1}
   164→              </div>
   165→              {idx < STATUS_WORKFLOW.length - 1 && (
   166→                <div className={cn('flex-1 h-0.5 min-w-[6px]', isActive ? 'bg-emerald-500' : 'bg-muted')} />
   167→              )}
   168→            </React.Fragment>
   169→          )
   170→        })}
   171→      </div>
   172→      <div className="flex flex-wrap gap-1 mt-2">
   173→        {STATUS_WORKFLOW.map((step, idx) => (
   174→          <span
   175→            key={step}
   176→            className={cn(
   177→              'text-[9px] px-1.5 py-0.5 rounded',
   178→              currentStep === idx
   179→                ? 'bg-emerald-100 text-emerald-700 dark:bg-emerald-900 dark:text-emerald-300 font-semibold'
   180→                : 'text-muted-foreground'
   181→            )}
   182→          >
   183→            {STATUS_CONFIG[step]?.label}
   184→          </span>
   185→        ))}
   186→      </div>
   187→    </div>
   188→  )
   189→}
