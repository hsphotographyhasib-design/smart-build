   103â†’// â”€â”€â”€ SLA Countdown Component â”€â”€â”€
   104â†’function SLACountdown({ deadline }: { deadline: string }) {
   105â†’  const [remaining, setRemaining] = useState('')
   106â†’
   107â†’  useEffect(() => {
   108â†’    function calc() {
   109â†’      const diff = new Date(deadline).getTime() - Date.now()
   110â†’      if (diff <= 0) {
   111â†’        setRemaining('BREACHED')
   112â†’        return
   113â†’      }
   114â†’      const hours = Math.floor(diff / 3600000)
   115â†’      const mins = Math.floor((diff % 3600000) / 60000)
   116â†’      if (hours > 24) {
   117â†’        const days = Math.floor(hours / 24)
   118â†’        setRemaining(`${days}d ${hours % 24}h`)
   119â†’      } else {
   120â†’        setRemaining(`${hours}h ${mins}m`)
   121â†’      }
   122â†’    }
   123â†’    calc()
   124â†’    const interval = setInterval(calc, 60000)
   125â†’    return () => clearInterval(interval)
   126â†’  }, [deadline])
   127â†’
   128â†’  const isBreached = remaining === 'BREACHED'
   129â†’
   130â†’  return (
   131â†’    <span className={cn(
   132â†’      'text-xs font-mono font-medium',
   133â†’      isBreached ? 'text-red-600 dark:text-red-400' : 'text-amber-600 dark:text-amber-400'
   134â†’    )}>
   135â†’      {isBreached && 'âš  '}{remaining}
   136â†’    </span>
   137â†’  )
   138â†’}
