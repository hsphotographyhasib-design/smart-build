   143→function OverviewTab({ projectId }: { projectId: string }) {
   144→  const { data: project, isLoading, error } = useQuery({
   145→    queryKey: ['project', projectId],
   146→    queryFn: () => api.get(`/api/projects/${projectId}`).then(r => r.data),
   147→    enabled: !!projectId,
   148→  })
   149→
   150→  const { data: insights } = useQuery({
   151→    queryKey: ['project-insights', projectId],
   152→    queryFn: () => api.get(`/api/projects/${projectId}/insights`).then(r => r.data),
   153→    enabled: !!projectId,
   154→  })
   155→
   156→  const { data: comments } = useQuery({
   157→    queryKey: ['project-comments', projectId],
   158→    queryFn: () => api.get(`/api/projects/${projectId}/comments`).then(r => r.data),
   159→    enabled: !!projectId,
   160→  })
   161→
   162→  if (isLoading) {
   163→    return (
   164→      <div className="space-y-6">
   165→        <KpiSkeleton />
   166→        <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
   167→          <Card><CardContent className="p-4"><Skeleton className="h-48 w-full" /></CardContent></Card>
   168→          <Card><CardContent className="p-4"><Skeleton className="h-48 w-full" /></CardContent></Card>
   169→        </div>
   170→      </div>
   171→    )
   172→  }
   173→
   174→  if (error) return <ErrorCard />
   175→
   176→  const p = project
   177→  const kpis = p?.kpis || {}
   178→  const budgetUsed = insights?.budgetUsed ?? kpis.totalExpenses ?? 0
   179→  const budgetTotal = p?.budget ?? 0
   180→  const budgetPct = budgetTotal > 0 ? Math.round((budgetUsed / budgetTotal) * 100) : 0
   181→  const commitmentsTotal = insights?.totalCommitments ?? 0
   182→  const openItemsCount = insights?.openItemsCount ?? 0
   183→  const pendingRfisCount = insights?.pendingRfisCount ?? 0
   184→  const teamSize = insights?.teamSize ?? p?.members?.length ?? 0
   185→  const daysLeft = p?.endDate ? differenceInDays(parseISO(p.endDate), new Date()) : null
   186→
   187→  return (
   188→    <div className="space-y-6">
   189→      {/* KPI Cards */}
   190→      <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-6 gap-4">
   191→        <Card>
   192→          <CardContent className="p-4">
   193→            <div className="flex items-center gap-2 text-muted-foreground text-xs mb-1"><DollarSign className="h-3.5 w-3.5" /> Budget</div>
   194→            <p className="text-lg font-semibold">{formatCurrency(budgetTotal)}</p>
   195→          </CardContent>
   196→        </Card>
   197→        <Card>
   198→          <CardContent className="p-4">
   199→            <div className="flex items-center gap-2 text-muted-foreground text-xs mb-1"><Target className="h-3.5 w-3.5" /> Progress</div>
   200→            <p className="text-lg font-semibold">{p?.progress ?? 0}%</p>
   201→          </CardContent>
   202→        </Card>
   203→        <Card>
   204→          <CardContent className="p-4">
   205→            <div className="flex items-center gap-2 text-muted-foreground text-xs mb-1"><Users className="h-3.5 w-3.5" /> Team</div>
   206→            <p className="text-lg font-semibold">{teamSize}</p>
   207→          </CardContent>
   208→        </Card>
   209→        <Card>
   210→          <CardContent className="p-4">
   211→            <div className="flex items-center gap-2 text-muted-foreground text-xs mb-1"><CircleDot className="h-3.5 w-3.5" /> Open Items</div>
   212→            <p className={cn('text-lg font-semibold', openItemsCount > 0 ? 'text-amber-600' : '')}>{openItemsCount}</p>
   213→          </CardContent>
   214→        </Card>
   215→        <Card>
   216→          <CardContent className="p-4">
   217→            <div className="flex items-center gap-2 text-muted-foreground text-xs mb-1"><MessageSquare className="h-3.5 w-3.5" /> Pending RFIs</div>
   218→            <p className={cn('text-lg font-semibold', pendingRfisCount > 0 ? 'text-orange-600' : '')}>{pendingRfisCount}</p>
   219→          </CardContent>
   220→        </Card>
   221→        <Card>
   222→          <CardContent className="p-4">
   223→            <div className="flex items-center gap-2 text-muted-foreground text-xs mb-1"><ShoppingCart className="h-3.5 w-3.5" /> Commitments</div>
   224→            <p className="text-lg font-semibold">{formatCurrency(commitmentsTotal)}</p>
   225→          </CardContent>
   226→        </Card>
   227→      </div>
   228→
   229→      <div className="grid grid-cols-1 lg:grid-cols-2 gap-4">
   230→        {/* Budget Health */}
   231→        <Card>
   232→          <CardHeader className="pb-3"><CardTitle className="text-sm font-medium">Budget Health</CardTitle></CardHeader>
   233→          <CardContent className="space-y-4">
   234→            <div className="space-y-2">
   235→              <div className="flex justify-between text-sm">
   236→                <span>Spent: {formatCurrency(budgetUsed)}</span>
   237→                <span className="text-muted-foreground">{budgetPct}%</span>
   238→              </div>
   239→              <Progress value={Math.min(budgetPct, 100)} className={cn('h-3', budgetPct > 90 ? '[&>div]:bg-red-500' : budgetPct > 70 ? '[&>div]:bg-amber-500' : '[&>div]:bg-emerald-500')} />
   240→            </div>
   241→            <div className="space-y-2">
   242→              <div className="flex justify-between text-sm">
   243→                <span>Committed: {formatCurrency(commitmentsTotal)}</span>
   244→              </div>
   245→              <div className="flex justify-between text-sm">
   246→                <span>Remaining: {formatCurrency(Math.max(0, budgetTotal - budgetUsed - commitmentsTotal))}</span>
   247→              </div>
   248→            </div>
   249→            {daysLeft !== null && (
   250→              <div className="flex items-center gap-2 text-sm text-muted-foreground pt-2 border-t">
   251→                <Clock className="h-4 w-4" />
   252→                <span>{daysLeft > 0 ? `${daysLeft} days remaining` : `${Math.abs(daysLeft)} days overdue`}</span>
   253→              </div>
   254→            )}
   255→          </CardContent>
   256→        </Card>
   257→
   258→        {/* Recent Activity */}
   259→        <Card>
   260→          <CardHeader className="pb-3"><CardTitle className="text-sm font-medium">Recent Activity</CardTitle></CardHeader>
   261→          <CardContent>
   262→            {comments && comments.length > 0 ? (
   263→              <ScrollArea className="max-h-64">
   264→                <div className="space-y-3">
   265→                  {comments.slice(0, 10).map((c: any) => (
   266→                    <div key={c.id} className="flex items-start gap-3">
   267→                      <div className="h-7 w-7 rounded-full bg-amber-100 flex items-center justify-center text-xs font-medium text-amber-700 shrink-0">
   268→                        {c.userName?.[0] || 'U'}
   269→                      </div>
   270→                      <div className="flex-1 min-w-0">
   271→                        <p className="text-sm"><span className="font-medium">{c.userName || 'User'}</span> <span className="text-muted-foreground">on {c.entityType}</span></p>
   272→                        <p className="text-xs text-muted-foreground truncate">{c.content}</p>
   273→                      </div>
   274→                      <span className="text-xs text-muted-foreground shrink-0">{c.createdAt ? format(parseISO(c.createdAt), 'dd MMM') : ''}</span>
   275→                    </div>
   276→                  ))}
   277→                </div>
   278→              </ScrollArea>
   279→            ) : (
   280→              <div className="flex flex-col items-center py-8 text-muted-foreground">
   281→                <Activity className="h-8 w-8 mb-2 opacity-40" />
   282→                <p className="text-sm">No recent activity</p>
   283→              </div>
   284→            )}
   285→          </CardContent>
   286→        </Card>
   287→      </div>
   288→    </div>
   289→  )
   290→}

   296→function TasksTab({ projectId }: { projectId: string }) {
   297→  const [open, setOpen] = useState(false)
   298→  const [form, setForm] = useState({ title: '', description: '', priority: 'medium', status: 'todo', startDate: '', endDate: '' })
   299→  const queryClient = useQueryClient()
   300→
   301→  const { data, isLoading, error } = useQuery({
   302→    queryKey: ['tasks', projectId],
   303→    queryFn: () => api.get(`/api/projects/${projectId}/tasks`).then(r => r.data),
   304→    enabled: !!projectId,
   305→  })
   306→
   307→  const mutation = useMutation({
   308→    mutationFn: (body: any) => api.post(`/api/projects/${projectId}/tasks`, body),
   309→    onSuccess: () => { toast.success('Task created'); queryClient.invalidateQueries({ queryKey: ['tasks', projectId] }); setOpen(false); setForm({ title: '', description: '', priority: 'medium', status: 'todo', startDate: '', endDate: '' }) },
   310→    onError: (e: any) => toast.error(e.error || 'Failed to create task'),
   311→  })
   312→
   313→  const tasks = Array.isArray(data) ? data : data?.tasks ?? []
   314→
   315→  if (isLoading) return <TableSkeleton />
   316→  if (error) return <ErrorCard />
   317→
   318→  return (
   319→    <div className="space-y-4">
   320→      <div className="flex items-center justify-between">
   321→        <div className="flex items-center gap-2">
   322→          <Search className="h-4 w-4 text-muted-foreground" />
   323→          <Input placeholder="Search tasks..." className="h-9 w-64" />
   324→        </div>
   325→        <Button size="sm" className="bg-amber-600 hover:bg-amber-700 text-white" onClick={() => setOpen(true)}>
   326→          <Plus className="h-4 w-4 mr-1" /> Add Task
   327→        </Button>
   328→      </div>
   329→
   330→      {tasks.length === 0 ? (
   331→        <EmptyCard icon={ListTodo} title="No tasks yet" description="Create your first task for this project."
   332→          action={<Button size="sm" variant="outline" onClick={() => setOpen(true)}><Plus className="h-4 w-4 mr-1" /> Add Task</Button>} />
   333→      ) : (
   334→        <Card>
   335→          <ScrollArea className="max-h-[600px]">
   336→            <Table>
   337→              <TableHeader>
   338→                <TableRow>
   339→                  <TableHead>Title</TableHead>
   340→                  <TableHead className="w-24">Status</TableHead>
   341→                  <TableHead className="w-24">Priority</TableHead>
   342→                  <TableHead className="hidden md:table-cell w-32">Assignee</TableHead>
   343→                  <TableHead className="hidden lg:table-cell w-28">Start</TableHead>
   344→                  <TableHead className="hidden lg:table-cell w-28">Due</TableHead>
   345→                  <TableHead className="w-20">Actions</TableHead>
   346→                </TableRow>
   347→              </TableHeader>
   348→              <TableBody>
   349→                {tasks.map((t: any) => (
   350→                  <TableRow key={t.id} className="hover:bg-amber-50/50 transition-colors">
   351→                    <TableCell className="font-medium">{t.title}</TableCell>
   352→                    <TableCell><Badge className={cn('text-xs', taskStatusColors[t.status] || 'bg-secondary')}>{t.status?.replace('_', ' ')}</Badge></TableCell>
   353→                    <TableCell><span className={cn('text-xs font-medium', priorityColors[t.priority])}>{t.priority}</span></TableCell>
   354→                    <TableCell className="hidden md:table-cell text-sm text-muted-foreground">{t.assignee?.name || '—'}</TableCell>
   355→                    <TableCell className="hidden lg:table-cell text-sm text-muted-foreground">{t.startDate ? format(parseISO(t.startDate), 'dd MMM yy') : '—'}</TableCell>
   356→                    <TableCell className="hidden lg:table-cell text-sm text-muted-foreground">{t.endDate ? format(parseISO(t.endDate), 'dd MMM yy') : '—'}</TableCell>
   357→                    <TableCell><Button variant="ghost" size="sm"><Eye className="h-4 w-4" /></Button></TableCell>
   358→                  </TableRow>
   359→                ))}
   360→              </TableBody>
   361→            </Table>
   362→          </ScrollArea>
   363→        </Card>
   364→      )}
   365→
   366→      <Dialog open={open} onOpenChange={setOpen}>
   367→        <DialogContent>
   368→          <DialogHeader><DialogTitle>Add Task</DialogTitle></DialogHeader>
   369→          <div className="space-y-3">
   370→            <div><Label>Title</Label><Input value={form.title} onChange={e => setForm({ ...form, title: e.target.value })} /></div>
   371→            <div><Label>Description</Label><Textarea value={form.description} onChange={e => setForm({ ...form, description: e.target.value })} rows={3} /></div>
   372→            <div className="grid grid-cols-2 gap-3">
   373→              <div><Label>Priority</Label><Select value={form.priority} onValueChange={v => setForm({ ...form, priority: v })}><SelectTrigger><SelectValue /></SelectTrigger><SelectContent><SelectItem value="low">Low</SelectItem><SelectItem value="medium">Medium</SelectItem><SelectItem value="high">High</SelectItem><SelectItem value="critical">Critical</SelectItem></SelectContent></Select></div>
   374→              <div><Label>Status</Label><Select value={form.status} onValueChange={v => setForm({ ...form, status: v })}><SelectTrigger><SelectValue /></SelectTrigger><SelectContent><SelectItem value="todo">To Do</SelectItem><SelectItem value="in_progress">In Progress</SelectItem><SelectItem value="completed">Completed</SelectItem></SelectContent></Select></div>
   375→            </div>
   376→            <div className="grid grid-cols-2 gap-3">
   377→              <div><Label>Start Date</Label><Input type="date" value={form.startDate} onChange={e => setForm({ ...form, startDate: e.target.value })} /></div>
   378→              <div><Label>Due Date</Label><Input type="date" value={form.endDate} onChange={e => setForm({ ...form, endDate: e.target.value })} /></div>
   379→            </div>
   380→          </div>
   381→          <DialogFooter>
   382→            <Button variant="outline" onClick={() => setOpen(false)}>Cancel</Button>
   383→            <Button className="bg-amber-600 hover:bg-amber-700 text-white" onClick={() => mutation.mutate(form)} disabled={mutation.isPending || !form.title}>
   384→              {mutation.isPending ? 'Creating...' : 'Create Task'}
   385→            </Button>
   386→          </DialogFooter>
   387→        </DialogContent>
   388→      </Dialog>
   389→    </div>
   390→  )
   391→}

   397→function TimelineTab({ projectId }: { projectId: string }) {
   398→  const { data: project, isLoading, error } = useQuery({
   399→    queryKey: ['project', projectId],
   400→    queryFn: () => api.get(`/api/projects/${projectId}`).then(r => r.data),
   401→    enabled: !!projectId,
   402→  })
   403→
   404→  if (isLoading) return <div className="space-y-4"><Skeleton className="h-4 w-48" /><div className="flex gap-6 overflow-x-auto pb-4">{Array.from({ length: 5 }).map((_, i) => <Skeleton key={i} className="h-28 w-48 shrink-0 rounded-lg" />)}</div></div>
   405→  if (error) return <ErrorCard />
   406→
   407→  const milestones = project?.milestones ?? []
   408→
   409→  if (milestones.length === 0) {
   410→    return <EmptyCard icon={Flag} title="No milestones" description="Milestones will appear here once they are added to the project." />
   411→  }
   412→
   413→  return (
   414→    <div className="space-y-6">
   415→      <div className="relative">
   416→        {/* Horizontal timeline line */}
   417→        <div className="absolute top-6 left-0 right-0 h-0.5 bg-amber-200 hidden md:block" />
   418→
   419→        <ScrollArea className="w-full">
   420→          <div className="flex gap-4 pb-4">
   421→            {milestones.map((m: any) => {
   422→              const isCompleted = m.status === 'completed'
   423→              const isOverdue = m.status === 'overdue'
   424→              return (
   425→                <Card key={m.id} className="shrink-0 w-48">
   426→                  <CardContent className="p-4 pt-6">
   427→                    {/* Dot on timeline */}
   428→                    <div className={cn(
   429→                      'h-3 w-3 rounded-full border-2 border-white shadow -mt-9 mb-3 mx-auto',
   430→                      isCompleted ? 'bg-emerald-500' : isOverdue ? 'bg-red-500' : 'bg-amber-500'
   431→                    )} />
